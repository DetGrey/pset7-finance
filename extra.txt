CREATE TABLE users (
	id INTEGER,
	username TEXT NOT NULL,
	hash TEXT NOT NULL,
	cash NUMERIC NOT NULL DEFAULT 10000.00,
	PRIMARY KEY(id));
CREATE UNIQUE INDEX username ON users (username);

CREATE TABLE bought (
	id INTEGER,
	symbol TEXT NOT NULL,
	shares INTEGER NOT NULL,
	price NUMERIC NOT NULL,
	transacted SMALLDATETIME,
	user_id INTEGER NOT NULL,
	PRIMARY KEY(id),
	FOREIGN KEY(user_id) REFERENCES users(id));




CREATE TABLE stocks (id INTEGER, name TEXT NOT NULL, symbol TEXT NOT NULL UNIQUE, shares INTEGER NOT NULL, price NUMERIC NOT NULL, total NUMERIC NOT NULL, user_id INTEGER NOT NULL, PRIMARY KEY(id), FOREIGN KEY(user_id) REFERENCES users(id));
